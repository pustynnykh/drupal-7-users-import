/**
 * Custom field processor for entityreference field type.
 */
function mymodule_user_import_default_field_processor(
    $user_fields, $field_name, $values
  ) {
  $field = $user_fields->$field_name;

  foreach ($values as $delta => $value) {
    if (!empty($value)) {
      $field[LANGUAGE_NONE][$delta]['target_id'] = $value;
    }
  }

  return $field;
}
